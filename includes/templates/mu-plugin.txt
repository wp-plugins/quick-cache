if (realpath(__FILE__) === realpath($_SERVER["SCRIPT_FILENAME"]))
	exit;
/*
Hide Quick Cache from plugins menu.
Attach to: add_filter("all_plugins");
Do NOT edit this file directly, it is re-built dynamically.
This prevents Quick Cache from being displayed as a plugin option
to anyone other than the Site Administrator operating on the main site.
*/
function ws_plugin__qcache_mu_hide_from_plugins_menu ($plugins = FALSE)
	{
		foreach ($plugins as $file => $plugin)
			if (preg_match("/quick-cache\.php$/", $file))
				unset($plugins[$file]);
		/**/
		return $plugins;
	}
/*
Add the filter. This prevents Quick Cache from being displayed as an option
to anyone other than the Site Administrator operating on the main site.
*/
if (function_exists("wpmu_current_site") && (int)$GLOBALS["blog_id"] > 1)
	{
		add_filter("all_plugins", "ws_plugin__qcache_mu_hide_from_plugins_menu");
	}
